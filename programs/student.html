C++ Student Personal Details Program:


#include <iostream>
#include <string>

using namespace std;

class Student {
private:
    int rollNo;
    string name;
    string className;
    string dob;       // Date of Birth
    string address;

public:
    // Constructor
    Student(int r, string n, string c, string d, string a) {
        rollNo = r;
        name = n;
        className = c;
        dob = d;
        address = a;
    }

    // Method to display the student's personal record
    void displayDetails() const {
        // Outputting with basic spacing
        cout << rollNo << "\t"
             << name << "\t"
             << className << "\t"
             << dob << "\t"
             << address << endl;
    }
};

int main() {
    cout << "--- C++ Student Personal Record System ---\\n\\n";

    // Create a simple array of Student objects
    Student students[2] = {
        Student(369, "Jatin", "B.Sc 3rd Sem", "2006-01-31", "Damra"),
        Student(206, "Shree", "B.Sc 3rd Sem", "2005-08-31", "Agia")
    };

    // Display the Header
    cout << "ROLL NO\tNAME\t\tCLASS\t\tDOB\t\tADDRESS\\n";
    cout << "---------------------------------------------------------------------------\\n";

    // Loop through the array and display each record
    for (int i = 0; i < 2; ++i) {
        students[i].displayDetails();
    }

    cout << "\\n---------------------------------------------------------------------------\\n";
    return 0;
}
