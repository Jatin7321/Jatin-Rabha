C++ Employee Record Program:


#include <iostream>
#include <string>

using namespace std;

// Class definition
class Employee {
private:
    int id;
    string name;
    string department;
    double salary;

public:
    // Constructor
    Employee(int emp_id, string emp_name, string emp_dept, double emp_salary) {
        id = emp_id;
        name = emp_name;
        department = emp_dept;
        salary = emp_salary;
    }

    // Method to display employee details
    void displayRecord() const {
        // Using basic spacing with cout
        cout << id << "\t\t" 
             << name << "\t\t"
             << department << "\t\t"
             << salary << endl;
    }
};

int main() {
    cout << "--- C++ Employee Record System (Simple) ---\\n\\n";

    // 1. Create a simple array of Employee objects
    Employee employees[3] = {
        Employee(101, "Alice Johnson", "HR", 55000.00),
        Employee(102, "Bob Smith", "Eng", 72500.50),
        Employee(103, "Charlie Brown", "Sales", 60000.00)
    };

    // 2. Display the Header
    cout << "ID\t\tNAME\t\tDEPARTMENT\tSALARY\\n";
    cout << "--------------------------------------------------------\\n";

    // 3. Loop through the array and display each record
    for (int i = 0; i < 3; ++i) {
        employees[i].displayRecord();
    }

    cout << "\\n--------------------------------------------------------\\n";
    return 0;
}
